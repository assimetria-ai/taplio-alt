### Option 2: Manual Control

```javascript
// routes/@custom/products.js
const express = require('express')
const router = express.Router()
const {
  pagination,
  handleList,
  handleGetById,
  handleCreate,
  parseQueryParams,
} = require('../../lib/@system/Helpers')

// List products
router.get('/api/products', pagination(), async (req, res, next) => {
  const config = parseQueryParams(req, {
    searchFields: ['name', 'description'],
    sortableFields: ['name', 'price', 'created_at'],
    filterFields: ['category', 'in_stock'],
  })
  
  await handleList({ repo: ProductRepo, req, res, next, filters: config })
})

// Get single product
router.get('/api/products/:id', async (req, res, next) => {
  await handleGetById({ repo: ProductRepo, req, res, next })
})

// Create product
router.post('/api/products', async (req, res, next) => {
  await handleCreate({ repo: ProductRepo, req, res, next })
})

module.exports = router
```
