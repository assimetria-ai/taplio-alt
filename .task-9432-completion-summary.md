## Usage Examples

### Quick Start - Dashboard
```jsx
import { DashboardLayout, StatCard, DataTable } from '@/app/components/@system/Dashboard'

function MyDashboard() {
  return (
    <DashboardLayout>
      <DashboardLayout.Content>
        <DashboardLayout.Header title="Dashboard" />
        <StatCard label="Users" value="1,234" />
        <DataTable columns={cols} data={data} searchable paginated />
      </DashboardLayout.Content>
    </DashboardLayout>
  )
}
```

### Quick Start - Onboarding
```jsx
import { OnboardingWizard } from '@/app/components/@system/Onboarding'

function OnboardingPage() {
  return (
    <OnboardingWizard
      onComplete={() => navigate('/app')}
      onSkip={() => navigate('/app')}
    />
  )
}
```

### Quick Start - Settings
```jsx
import {
  UserSettings,
  ProfileSettings,
  SecuritySettings,
} from '@/app/components/@system/UserSettings'

function SettingsPage() {
  return (
    <UserSettings defaultTab="profile">
      <UserSettings.Tab id="profile" label="Profile">
        <ProfileSettings />
      </UserSettings.Tab>
      <UserSettings.Tab id="security" label="Security">
        <SecuritySettings />
      </UserSettings.Tab>
    </UserSettings>
  )
}
```
