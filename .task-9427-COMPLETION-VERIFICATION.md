```javascript
async function blacklistAccessToken(token) {
```

**Password reset flow:**

```javascript
// Step 1: Request reset
router.post('/password/request', passwordResetLimiter, validate(PasswordResetRequestBody), async (req, res) => {
  const { email } = req.body
  const user = await UserRepo.findByEmail(email.trim().toLowerCase())
  
  if (user) {
    const token = await createPasswordResetToken(user.id)
    await emailService.sendPasswordResetEmail(user.email, user.name, token)
  }
  
  // Always return success (anti-enumeration)
  res.json({ message: 'If that email exists, a reset link has been sent.' })
})

// Step 2: Reset with token
router.post('/password/reset', passwordResetLimiter, validate(PasswordResetBody), async (req, res, next) => {
  const { token, password } = req.body
  
  // Validate token and get user
  const tokenRecord = await db.oneOrNone(
    'SELECT user_id FROM password_reset_tokens WHERE token_hash = $1 AND used_at IS NULL AND expires_at > now()',
    [hashToken(token)]
  )
  
  if (!tokenRecord) {
    return res.status(400).json({ error: 'Invalid or expired reset token.' })
  }
  
  // Update password
  const passwordHash = await bcrypt.hash(password, 12)
  await UserRepo.updatePassword(tokenRecord.user_id, passwordHash)
  
  // Mark token as used
  await db.none('UPDATE password_reset_tokens SET used_at = now() WHERE token_hash = $1', [hashToken(token)])
  
  res.json({ message: 'Password reset successfully.' })
})
```

```javascript
  const rawUrl = process.env.APP_URL ?? 'http://localhost:5173'
  
  const url = new URL(rawUrl)
  
  // SECURITY: Only allow http/https protocols
  if (!['http:', 'https:'].includes(url.protocol)) {
    throw new Error(`Invalid APP_URL protocol: ${url.protocol}`)
  }
  
  // Prevent suspicious URLs
  if (url.href !== rawUrl) {
```

**OAuth flow:**
```javascript
// Step 1: Initiate
router.get('/google', oauthLimiter, (req, res) => {
  const state = crypto.randomBytes(16).toString('hex')
  req.session.oauthState = state
  
  const authUrl = google.getAuthorizationUrl(state)
  res.redirect(authUrl)
})

// Step 2: Callback
router.get('/google/callback', oauthLimiter, async (req, res, next) => {
  const { code, state } = req.query
  
  // Validate state (CSRF protection)
  if (!state || state !== req.session.oauthState) {
    return res.redirect(`${appUrl()}/auth?error=oauth_failed`)
  }
  
  // Exchange code for tokens
  const tokens = await google.exchangeCode(code)
  const profile = await google.getUserProfile(tokens.access_token)
  
  // Find or create user
  let user = await UserRepo.findByEmail(profile.email)
  if (!user) {
    user = await UserRepo.create({
      name: profile.name,
      email: profile.email,
      email_verified_at: new Date(),
    })
  }
  
  // Link OAuth account
  await OAuthRepo.upsert({
    user_id: user.id,
    provider: 'google',
    provider_user_id: profile.id,
    access_token: tokens.access_token,
  })
  
  // Create session
  const jwt = await signTokenAsync({ userId: user.id, email: user.email })
  res.cookie('session_token', jwt, { httpOnly: true, secure: true, maxAge: SESSION_TTL * 1000 })
  
  res.redirect(`${appUrl()}/app`)
})
```
