function canAssignTask(taskId) {

    console.log(`Task #${taskId} already completed`)
    return false
  }
  
  // Check database status

  if (task.status === 'completed') {
    console.log(`Task #${taskId} marked complete in DB`)
    return false
  }
  
  // Check recent assignment history

    return false
  }
  
  return true
}
