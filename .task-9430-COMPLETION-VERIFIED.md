## Usage Example

```javascript
// Auto-generated CRUD router (simplest approach)
const { createCrudRouter } = require('../../lib/@system/Helpers')
const ProductRepo = require('../../db/repos/ProductRepo')

module.exports = createCrudRouter({
  repo: ProductRepo,
  validation: { create: CreateSchema, update: UpdateSchema },
  config: { basePath: '/api/products', dataKey: 'product' }
})

//          POST /api/products, PATCH /api/products/:id,
```
