```javascript
function canAssignTask(taskId) {
  // 1. Check git log for completion commits
  const commits = execSync(`git log --oneline --grep="${taskId}"`).toString()
  if (commits.includes('feat():')) return false
  
  // 2. Check for completion markers
  const completionFiles = glob.sync(`.task-${taskId}-*completion*.md`)
  if (completionFiles.length > 0) return false
  
  // 3. Check assignment count
  const assignmentCount = db.query('SELECT COUNT(*) FROM task_assignments WHERE task_id = ?', [taskId])
  if (assignmentCount > 2) {
    throw new Error(`Task ${taskId} has been assigned ${assignmentCount} times - manual review required`)
  }
  
  return true
}
```
