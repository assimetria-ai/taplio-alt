```javascript
const { createCrudRouter } = require('./lib/@system/Helpers')

// Zero boilerplate - complete REST API
const router = createCrudRouter({
  repo: ProductRepo,
  config: { basePath: '/api/products' }
})
// Creates: GET /list, GET /:id, POST /, PATCH /:id, DELETE /:id
```

### 2. Pagination

```javascript
pagination({ defaultLimit: 20, maxLimit: 100 })
// Supports: ?page=2&limit=20 or ?offset=40&limit=20
```

### 3. Sorting

```javascript
sorting({ allowedFields: ['name', 'price'] })
// Supports: ?sort=price&order=asc
// Or multi-field: ?sort=category:asc,price:desc
```

### 4. Filtering

```javascript
filtering({ allowedFields: ['category', 'in_stock'] })
// Basic: ?category=electronics&in_stock=true

advancedFiltering({ allowedFields: ['price', 'rating'] })
// Advanced: ?price[gte]=100&price[lte]=500&rating[gt]=4
```
