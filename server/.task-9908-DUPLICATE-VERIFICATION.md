## Complete Example

Here's a working example using all features:

```javascript
const express = require('express')
const router = express.Router()
const {
  pagination,
  sorting,
  filtering,
  handleList,
  handleGetById,
  handleCreate,
  handleUpdate,
  handleDelete,
  authenticate,
} = require('../../lib/@system/Helpers')
const ProductRepo = require('../../db/repos/ProductRepo')

// List products with pagination, sorting, and filtering
router.get('/api/products',
  pagination({ defaultLimit: 20, maxLimit: 100 }),
  sorting({ 
    allowedFields: ['name', 'price', 'created_at'],
    defaultField: 'created_at',
    defaultOrder: 'desc'
  }),
  filtering({
    allowedFields: ['category', 'in_stock', 'brand'],
    booleanFields: ['in_stock']
  }),
  async (req, res, next) => {
    await handleList({
      repo: ProductRepo,
      req, res, next,
      filters: {
        whereClause: req.filters.whereClause,
        params: req.filters.params,
        orderBy: req.sorting.clause
      }
    })
  }
)

// Get single product
router.get('/api/products/:id',
  async (req, res, next) => {
    await handleGetById({
      repo: ProductRepo,
      req, res, next,
      dataKey: 'product'
    })
  }
)

// Create product (protected)
router.post('/api/products', authenticate,
  async (req, res, next) => {
    await handleCreate({
      repo: ProductRepo,
      req, res, next,
      dataKey: 'product',
      validate: CreateProductSchema
    })
  }
)

module.exports = router
```

**Query examples:**
- `GET /api/products?page=2&limit=20&sort=price:asc&category=electronics&in_stock=true`
- `GET /api/products?price[gte]=100&price[lte]=500&sort=price:asc,name:asc`
