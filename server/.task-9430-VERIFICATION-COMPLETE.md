## Usage Example

```javascript
const { createCrudRouter } = require('../../lib/@system/Helpers')
const ProductRepo = require('../../db/repos/ProductRepo')

// One line creates complete REST API
module.exports = createCrudRouter({
  repo: ProductRepo,
  config: { basePath: '/api/products', dataKey: 'product' }
})

// Automatically creates:
// GET    /api/products       - List with pagination/search/filter
// GET    /api/products/:id   - Get by ID
// POST   /api/products       - Create
// PATCH  /api/products/:id   - Update
```
