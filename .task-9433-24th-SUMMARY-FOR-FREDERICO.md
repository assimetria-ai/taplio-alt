### Pre-Assignment Check (Pseudocode)

```javascript
function canAssignTask(taskId, assignToAgent) {
  // Check git history
  const completionCommits = gitLog(`--grep="task #${taskId}.*feat()"`)
  if (completionCommits.length > 0) {
    console.error(`Task #${taskId} already complete: ${completionCommits}`)
    return false
  }
  
  // Check assignment count
  const assignments = db.query(`SELECT COUNT(*) FROM task_assignments WHERE task_id = ${taskId}`)
  if (assignments.count >= 2) {
    console.warn(`Task #${taskId} assigned ${assignments.count} times already!`)
    requireManualApproval()
  }
  
  // Check for completion markers
  const completionFiles = fs.glob(`.task-${taskId}-*-completion.md`)
  if (completionFiles.length > 0) {
    console.error(`Task #${taskId} has completion markers: ${completionFiles}`)
    return false
  }
  
  return true
}
```

function onGitPush(commits) {
  commits.forEach(commit => {
    // Parse commit message: "feat(): task #9433 - Mobile responsiveness"
    const taskMatch = commit.message.match(/task #(\d+)/)
    const isFeat = commit.message.startsWith('feat(')
    
    if (taskMatch && isFeat) {
      const taskId = taskMatch[1]
      
      // Update database

        status: 'COMPLETED',

        completed_by: commit.author,
        completion_commit: commit.hash
      }, { id: taskId })
      
      console.log(`✅ Auto-marked task #${taskId} as COMPLETED`)
    }
  })
}
