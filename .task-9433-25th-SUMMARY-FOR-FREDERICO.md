**Implement pre-assignment validation:**

```javascript
async function validateTaskAssignment(taskId) {
  // 1. Check git history
  const commits = await exec(`git log --oneline --grep="task #${taskId}"`)
  if (commits.includes('feat():')) {
    throw new Error(`Task ${taskId} already has feature commit - likely complete`)
  }
  
  // 2. Check assignment count
  const count = await db.query(
    'SELECT COUNT(*) as count FROM task_assignments WHERE task_id = ?',
    [taskId]
  )
  
  if (count.count >= 3) {
    throw new Error(
      `Task ${taskId} assigned ${count.count} times - manual review required`
    )
  }
  
  // 3. Check completion markers
  const completionFiles = await glob(`.task-${taskId}-*completion*.md`)
  if (completionFiles.length > 0) {
    throw new Error(`Task ${taskId} has completion files - likely finished`)
  }
  
  return true
}
```
